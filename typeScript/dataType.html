JavaScript зҡ„зұ»еһӢеҲҶдёәдёӨз§ҚпјҡеҺҹе§Ӣж•°жҚ®зұ»еһӢпјҲPrimitive data typesпјүе’ҢеҜ№иұЎзұ»еһӢпјҲObject typesпјүгҖӮ

еҺҹе§Ӣж•°жҚ®зұ»еһӢеҢ…жӢ¬пјҡеёғе°”еҖјгҖҒж•°еҖјгҖҒеӯ—з¬ҰдёІгҖҒnullгҖҒundefined д»ҘеҸҠ ES6 дёӯзҡ„ж–°зұ»еһӢ SymbolгҖӮ

TypeScriptж”ҜжҢҒдёҺJavaScriptеҮ д№ҺзӣёеҗҢзҡ„ж•°жҚ®зұ»еһӢпјҢжӯӨеӨ–иҝҳжҸҗдҫӣдәҶе®һз”Ёзҡ„жһҡдёҫзұ»еһӢж–№дҫҝжҲ‘д»¬дҪҝз”ЁгҖӮ

# еҺҹе§Ӣж•°жҚ®зұ»еһӢ

# еёғе°”еҖј boolean

жңҖеҹәжң¬зҡ„ж•°жҚ®зұ»еһӢе°ұжҳҜз®ҖеҚ•зҡ„true/falseеҖјпјҢеңЁJavaScriptе’ҢTypeScriptйҮҢеҸ«еҒҡ boolean(е…¶д»–иҜӯиЁҖдёӯд№ҹдёҖж ·)

let isDone:boolean = false;

# ж•°еҖј number

е’ҢJavaScriptдёҖж ·пјҢTypeScriptйҮҢзҡ„жүҖжңүж•°еӯ—йғҪжҳҜжө®зӮ№ж•°гҖӮиҝҷдәӣжө®зӮ№ж•°зҡ„зұ»еһӢжҳҜnumberгҖӮйҷӨдәҶж”ҜжҢҒеҚҒиҝӣеҲ¶е’ҢеҚҒе…ӯиҝӣеҲ¶еӯ—йқўйҮҸпјҢTypeScriptиҝҳж”ҜжҢҒECMAScript 2015дёӯеј•е…Ҙзҡ„дәҢиҝӣеҲ¶е’Ңе…«иҝӣеҲ¶еӯ—йқўйҮҸгҖӮ

let decLiteral:number = 20; //еҚҒиҝӣеҲ¶
let hexLiteral:number = 0x14; //еҚҒе…ӯиҝӣеҲ¶
let binaryLiteral:number = 0b10100; //дәҢиҝӣеҲ¶
let octalLiteral:number = 0o24; //е…«иҝӣеҲ¶

# еӯ—з¬ҰдёІ string

JavaScript зЁӢеәҸзҡ„еҸҰдёҖйЎ№еҹәжң¬ж“ҚдҪңжҳҜеӨ„зҗҶзҪ‘йЎөжҲ–жңҚеҠЎеҷЁз«Ҝзҡ„ж–Үжң¬ж•°жҚ®гҖӮ еғҸе…¶е®ғиҜӯиЁҖйҮҢдёҖж ·пјҢжҲ‘д»¬дҪҝз”Ё string иЎЁзӨәж–Үжң¬ж•°жҚ®зұ»еһӢгҖӮ е’Ң JavaScript дёҖж ·пјҢеҸҜд»ҘдҪҝз”ЁеҸҢеј•еҸ·пјҲ"пјүжҲ–еҚ•еј•еҸ·пјҲ'пјүиЎЁзӨәеӯ—з¬ҰдёІгҖӮ

let name:string = "еј дёү"
let sentence = `hello,${name}`

# nullе’Ңundefined

null е’Ң undefinedжң¬иә«жҳҜдёҖдёӘеҖјпјҢеҪ“еҒҡзұ»еһӢи®Іж—¶пјҢnullе’ҢundefinedжҳҜжүҖжңүзұ»еһӢзҡ„еӯҗзұ»еһӢ,еӯҗзұ»еһӢжҳҜеҸҜд»ҘиөӢеҖјз»ҷзҲ¶зұ»еһӢзҡ„гҖӮ

nullе’ҢundefinedжҳҜе…¶д»–зұ»еһӢзҡ„еӯҗзұ»еһӢ

let u: undefined = undefined;
let n: null = null;

let a:undefined = null;

let num:number | null = 3;//иҒ”еҗҲзұ»еһӢ
num = null;

дёҺ void зҡ„еҢәеҲ«жҳҜпјҢundefined е’Ң null жҳҜжүҖжңүзұ»еһӢзҡ„еӯҗзұ»еһӢгҖӮд№ҹе°ұжҳҜиҜҙ undefined зұ»еһӢзҡ„еҸҳйҮҸпјҢеҸҜд»ҘиөӢеҖјз»ҷ number зұ»еһӢзҡ„еҸҳйҮҸпјҡ

// иҝҷж ·дёҚдјҡжҠҘй”ҷ
let num: number = undefined;

// иҝҷж ·д№ҹдёҚдјҡжҠҘй”ҷ
let u: undefined;
let num: number = u;

иҖҢ void зұ»еһӢзҡ„еҸҳйҮҸдёҚиғҪиөӢеҖјз»ҷ number зұ»еһӢзҡ„еҸҳйҮҸпјҡ

let u: void;
let num: number = u;
вҖӢ
// Type 'void' is not assignable to type 'number'.

# ж•°з»„

TypeScript еғҸ JavaScript дёҖж ·еҸҜд»Ҙж“ҚдҪңж•°з»„е…ғзҙ гҖӮ жңү3з§Қж–№ејҸеҸҜд»Ҙе®ҡд№үж•°з»„гҖӮ з¬¬дёҖз§ҚпјҢеҸҜд»ҘеңЁе…ғзҙ зұ»еһӢеҗҺйқўжҺҘдёҠ []пјҢиЎЁзӨәз”ұжӯӨзұ»еһӢе…ғзҙ з»„жҲҗзҡ„дёҖдёӘж•°з»„пјҡ

let list: number[] = [1, 2, 3]
var arr:number[] =[1,2,3]
var arr1:string[]=["1","2"]
var arr2:any[] = [1,"2",true]

з¬¬дәҢз§Қж–№ејҸжҳҜдҪҝз”Ёж•°з»„жіӣеһӢпјҢArray<е…ғзҙ зұ»еһӢ>пјҡ

let list: Array<number> = [1, 2, 3]
var arrType:Array<number> = [1,2,3]
var arrType2:Array<string> =["1","2"]
var arrType3:Array<any> =[1,"2",true]

з¬¬дёүз§Қж–№ејҸеҸҜйҮҮз”ЁжҺҘеҸЈиЎЁзӨәжі•

//жҺҘеҸЈиЎЁзӨә
interface Istate {
    username:string,
    age:number
}
interface IArr {
    //[index:number]:number
    [index:number]:Istate
}

//var arrType4:IArr =[1,2,3]
var arrType4:IArr =[{username:"еј дёү",age:10}]
//жӯӨж—¶пјҢејәеҲ¶зәҰе®ҡж•°з»„йҮҢзҡ„жҜҸдёҖйЎ№еҝ…йЎ»жҳҜз¬ҰеҗҲIstateжҺҘеҸЈзҡ„зәҰжқҹи§„еҲҷ

//еҗҢзҗҶ
var arrType5:Array<Istate> = [{username:"еј дёү",age:10}]

var arrType6:Istate[] = [{username:"еј дёү",age:10}]

# е…ғз»„ Tuple

е…ғз»„зұ»еһӢе…Ғи®ёиЎЁзӨәдёҖдёӘе·ІзҹҘе…ғзҙ ж•°йҮҸе’Ңзұ»еһӢзҡ„ж•°з»„пјҢеҗ„е…ғзҙ зҡ„зұ»еһӢдёҚеҝ…зӣёеҗҢгҖӮ жҜ”еҰӮпјҢдҪ еҸҜд»Ҙе®ҡд№үдёҖеҜ№еҖјеҲҶеҲ«дёә string е’Ң number зұ»еһӢзҡ„е…ғз»„гҖӮ

let x: [string, number]
x = ['hello', 10] // OK
x = [10, 'hello'] // Error

let  arr:[string,number] = ["aaa",10]

еҪ“и®ҝй—®дёҖдёӘе·ІзҹҘзҙўеј•зҡ„е…ғзҙ пјҢдјҡеҫ—еҲ°жӯЈзЎ®зҡ„зұ»еһӢпјҡ

console.log(x[0].substr(1)) // OK
console.log(x[1].substr(1)) // Error, 'number' дёҚеӯҳеңЁ 'substr' ж–№жі•

еҪ“и®ҝй—®дёҖдёӘи¶Ҡз•Ңзҡ„е…ғзҙ пјҢдјҡдҪҝз”ЁиҒ”еҗҲзұ»еһӢжӣҝд»Јпјҡ

x[3] = 'world' // OK, еӯ—з¬ҰдёІеҸҜд»ҘиөӢеҖјз»ҷ(string | number)зұ»еһӢ

console.log(x[5].toString()) // OK, 'string' е’Ң 'number' йғҪжңү toString

x[6] = true // Error, еёғе°”дёҚжҳҜ(string | number)зұ»еһӢ

жіЁж„ҸпјҡиҮӘд»Һ TyeScript 3.1 зүҲжң¬д№ӢеҗҺпјҢи®ҝй—®и¶Ҡз•Ңе…ғзҙ дјҡжҠҘй”ҷпјҢжҲ‘д»¬дёҚеә”иҜҘеҶҚдҪҝз”ЁиҜҘзү№жҖ§гҖӮ

# д»»ж„ҸеҖј any

жңүж—¶еҖҷпјҢжҲ‘д»¬дјҡжғіиҰҒдёәйӮЈдәӣеңЁзј–зЁӢйҳ¶ж®өиҝҳдёҚжё…жҘҡзұ»еһӢзҡ„еҸҳйҮҸжҢҮе®ҡдёҖдёӘзұ»еһӢгҖӮ иҝҷдәӣеҖјеҸҜиғҪжқҘиҮӘдәҺеҠЁжҖҒзҡ„еҶ…е®№пјҢжҜ”еҰӮжқҘиҮӘз”ЁжҲ·иҫ“е…ҘжҲ–з¬¬дёүж–№д»Јз Ғеә“гҖӮ иҝҷз§Қжғ…еҶөдёӢпјҢжҲ‘д»¬дёҚеёҢжңӣзұ»еһӢжЈҖжҹҘеҷЁеҜ№иҝҷдәӣеҖјиҝӣиЎҢжЈҖжҹҘиҖҢжҳҜзӣҙжҺҘи®©е®ғд»¬йҖҡиҝҮзј–иҜ‘йҳ¶ж®өзҡ„жЈҖжҹҘгҖӮ йӮЈд№ҲжҲ‘д»¬еҸҜд»ҘдҪҝз”Ё any зұ»еһӢжқҘж Үи®°иҝҷдәӣеҸҳйҮҸпјҡ

д»»ж„ҸеҖјпјҲAnyпјүз”ЁжқҘиЎЁзӨәе…Ғи®ёиөӢеҖјдёәд»»ж„Ҹзұ»еһӢ

еЈ°жҳҺдёҖдёӘеҸҳйҮҸдёәд»»ж„ҸеҖјд№ӢеҗҺпјҢеҜ№е®ғзҡ„д»»дҪ•ж“ҚдҪңпјҢиҝ”еӣһзҡ„еҶ…е®№зҡ„зұ»еһӢйғҪжҳҜд»»ж„ҸеҖјгҖӮ

let notSure: any = 4
notSure = 'maybe a string instead'
notSure = false // д№ҹеҸҜд»ҘжҳҜдёӘ boolean

еңЁеҜ№зҺ°жңүд»Јз ҒиҝӣиЎҢж”№еҶҷзҡ„ж—¶еҖҷпјҢany зұ»еһӢжҳҜеҚҒеҲҶжңүз”Ёзҡ„пјҢе®ғе…Ғи®ёдҪ еңЁзј–иҜ‘ж—¶еҸҜйҖүжӢ©ең°еҢ…еҗ«жҲ–з§»йҷӨзұ»еһӢжЈҖжҹҘгҖӮе№¶дё”еҪ“дҪ еҸӘзҹҘйҒ“дёҖйғЁеҲҶж•°жҚ®зҡ„зұ»еһӢж—¶пјҢany зұ»еһӢд№ҹжҳҜжңүз”Ёзҡ„гҖӮ жҜ”еҰӮпјҢдҪ жңүдёҖдёӘж•°з»„пјҢе®ғеҢ…еҗ«дәҶдёҚеҗҢзҡ„зұ»еһӢзҡ„ж•°жҚ®пјҡ

let list: any[] = [1, true, 'free']

list[1] = 100

еҸҳйҮҸеҰӮжһңеңЁеЈ°жҳҺзҡ„ж—¶еҖҷпјҢжңӘжҢҮе®ҡе…¶зұ»еһӢпјҢйӮЈд№Ҳе®ғдјҡиў«иҜҶеҲ«дёәд»»ж„ҸеҖјзұ»еһӢгҖӮ

var num2; //жІЎжңүиөӢеҖјж“ҚдҪңпјҢе°ұдјҡиў«и®Өдёәд»»ж„ҸеҖјзұ»еһӢ
num2 = true;
num2 = "3";

# з©әеҖј void

жҹҗз§ҚзЁӢеәҰдёҠжқҘиҜҙпјҢvoid зұ»еһӢеғҸжҳҜдёҺ any зұ»еһӢзӣёеҸҚпјҢе®ғиЎЁзӨәжІЎжңүд»»дҪ•зұ»еһӢгҖӮ еҪ“дёҖдёӘеҮҪж•°жІЎжңүиҝ”еӣһеҖјж—¶пјҢдҪ йҖҡеёёдјҡи§ҒеҲ°е…¶иҝ”еӣһеҖјзұ»еһӢжҳҜ voidпјҡ

function warnUser(): void {
  console.log('This is my warning message')
}

еЈ°жҳҺдёҖдёӘ void зұ»еһӢзҡ„еҸҳйҮҸжІЎжңүд»Җд№ҲеӨ§з”ЁпјҢеӣ дёәдҪ еҸӘиғҪдёәе®ғиөӢдәҲ undefined е’Ң nullпјҡ

let unusable: void = undefined

# жһҡдёҫ enum

жһҡдёҫпјҲenumпјүзұ»еһӢз”ЁдәҺеҸ–еҖјиў«йҷҗе®ҡеңЁдёҖе®ҡиҢғеӣҙеҶ…зҡ„еңәжҷҜ

йҮҮз”Ёе…ій”®еӯ—enumе®ҡд№ү

дҫӢеҰӮпјҡenum Days{Sun, Mon,Tue}

жһҡдёҫжҲҗе‘ҳдјҡиў«иөӢеҖјдёәд»Һ0ејҖе§ӢйҖ’еўһзҡ„ж•°еӯ—пјҢеҗҢж—¶д№ҹдјҡиў«жһҡдёҫеҖјеҲ°жһҡдёҫеҗҚиҝӣиЎҢеҸҚеҗ‘жҳ е°„гҖӮ

enum зұ»еһӢжҳҜеҜ№ JavaScript ж ҮеҮҶж•°жҚ®зұ»еһӢзҡ„дёҖдёӘиЎҘе……гҖӮ еғҸ C# зӯүе…¶е®ғиҜӯиЁҖдёҖж ·пјҢдҪҝз”Ёжһҡдёҫзұ»еһӢеҸҜд»ҘдёәдёҖз»„ж•°еҖјиөӢдәҲеҸӢеҘҪзҡ„еҗҚеӯ—гҖӮ

enum Color {Red, Green, Blue}
let c: Color = Color.Green

й»ҳи®Өжғ…еҶөдёӢпјҢд»Һ 0 ејҖе§Ӣдёәе…ғзҙ зј–еҸ·гҖӮ дҪ д№ҹеҸҜд»ҘжүӢеҠЁзҡ„жҢҮе®ҡжҲҗе‘ҳзҡ„ж•°еҖјгҖӮ дҫӢеҰӮпјҢжҲ‘д»¬е°ҶдёҠйқўзҡ„дҫӢеӯҗж”№жҲҗд»Һ 1 ејҖе§Ӣзј–еҸ·пјҡ

enum Color {Red = 1, Green, Blue}
let c: Color = Color.Green

жҲ–иҖ…пјҢе…ЁйғЁйғҪйҮҮз”ЁжүӢеҠЁиөӢеҖјпјҡ

enum Color {Red = 1, Green = 2, Blue = 4}
let c: Color = Color.Green

жһҡдёҫзұ»еһӢжҸҗдҫӣзҡ„дёҖдёӘдҫҝеҲ©жҳҜдҪ еҸҜд»Ҙз”ұжһҡдёҫзҡ„еҖјеҫ—еҲ°е®ғзҡ„еҗҚеӯ—гҖӮ дҫӢеҰӮпјҢжҲ‘д»¬зҹҘйҒ“ж•°еҖјдёә 2пјҢдҪҶжҳҜдёҚзЎ®е®ҡе®ғжҳ е°„еҲ° Color йҮҢзҡ„е“ӘдёӘеҗҚеӯ—пјҢжҲ‘д»¬еҸҜд»ҘжҹҘжүҫзӣёеә”зҡ„еҗҚеӯ—пјҡ

enum Color {Red = 1, Green, Blue}
let colorName: string = Color[2]

console.log(colorName)  // жҳҫзӨә'Green'еӣ дёәдёҠйқўд»Јз ҒйҮҢе®ғзҡ„еҖјжҳҜ2

//дҪҝз”ЁжһҡдёҫеҸҜд»Ҙе®ҡд№үдёҖдәӣжңүеҗҚеӯ—зҡ„ж•°еӯ—еёёйҮҸ
enum Days{
    Sun,
    Mon,
    Tue,
    Wed,
    Thu,
    Fri,
    Sat
}

console.log(Days.Sun)//0
console.log(Days.Sat)//6
//жһҡдёҫзұ»еһӢдјҡиў«зј–иҜ‘жҲҗдёҖдёӘеҸҢеҗ‘жҳ е°„зҡ„еҜ№иұЎ
console.log(Days[0])//Sun

let day:Days= Days.Sun;

# е…¶д»–зұ»еһӢ never

never зұ»еһӢиЎЁзӨәзҡ„жҳҜйӮЈдәӣж°ёдёҚеӯҳеңЁзҡ„еҖјзҡ„зұ»еһӢгҖӮ дҫӢеҰӮпјҢ never зұ»еһӢжҳҜйӮЈдәӣжҖ»жҳҜдјҡжҠӣеҮәејӮеёёжҲ–ж №жң¬е°ұдёҚдјҡжңүиҝ”еӣһеҖјзҡ„еҮҪж•°иЎЁиҫҫејҸжҲ–з®ӯеӨҙеҮҪж•°иЎЁиҫҫејҸзҡ„иҝ”еӣһеҖјзұ»еһӢпјӣ еҸҳйҮҸд№ҹеҸҜиғҪжҳҜ never зұ»еһӢпјҢеҪ“е®ғд»¬иў«ж°ёдёҚдёәзңҹзҡ„зұ»еһӢдҝқжҠӨжүҖзәҰжқҹж—¶гҖӮ

never зұ»еһӢжҳҜд»»дҪ•зұ»еһӢзҡ„еӯҗзұ»еһӢпјҢд№ҹеҸҜд»ҘиөӢеҖјз»ҷд»»дҪ•зұ»еһӢпјӣз„¶иҖҢпјҢжІЎжңүзұ»еһӢжҳҜ never зҡ„еӯҗзұ»еһӢжҲ–еҸҜд»ҘиөӢеҖјз»ҷnever зұ»еһӢпјҲйҷӨдәҶ never жң¬иә«д№ӢеӨ–пјүгҖӮ еҚідҪҝ any д№ҹдёҚеҸҜд»ҘиөӢеҖјз»ҷ neverгҖӮ

дёӢйқўжҳҜдёҖдәӣиҝ”еӣһ never зұ»еһӢзҡ„еҮҪж•°пјҡ

// иҝ”еӣһneverзҡ„еҮҪж•°еҝ…йЎ»еӯҳеңЁж— жі•иҫҫеҲ°зҡ„з»ҲзӮ№
function error(message: string): never {
  throw new Error(message)
}

// жҺЁж–ӯзҡ„иҝ”еӣһеҖјзұ»еһӢдёәnever
function fail() {
  return error("Something failed")
}

// иҝ”еӣһneverзҡ„еҮҪж•°еҝ…йЎ»еӯҳеңЁж— жі•иҫҫеҲ°зҡ„з»ҲзӮ№
function infiniteLoop(): never {
  while (true) {
  }
}

# object

object иЎЁзӨәйқһеҺҹе§Ӣзұ»еһӢпјҢд№ҹе°ұжҳҜйҷӨ numberпјҢstringпјҢbooleanпјҢsymbolпјҢnullжҲ–undefined д№ӢеӨ–зҡ„зұ»еһӢгҖӮ

дҪҝз”Ё object зұ»еһӢпјҢе°ұеҸҜд»ҘжӣҙеҘҪзҡ„иЎЁзӨәеғҸ Object.create иҝҷж ·зҡ„ APIгҖӮдҫӢеҰӮпјҡ

declare function create(o: object | null): void

create({ prop: 0 }) // OK
create(null) // OK

create(42) // Error
create('string') // Error
create(false) // Error
create(undefined) // Error

# зұ»еһӢж–ӯиЁҖ

жңүж—¶еҖҷдҪ дјҡйҒҮеҲ°иҝҷж ·зҡ„жғ…еҶөпјҢдҪ дјҡжҜ” TypeScript жӣҙдәҶи§ЈжҹҗдёӘеҖјзҡ„иҜҰз»ҶдҝЎжҒҜгҖӮ йҖҡеёёиҝҷдјҡеҸ‘з”ҹеңЁдҪ жё…жҘҡең°зҹҘйҒ“дёҖдёӘе®һдҪ“е…·жңүжҜ”е®ғзҺ°жңүзұ»еһӢжӣҙзЎ®еҲҮзҡ„зұ»еһӢгҖӮ

йҖҡиҝҮзұ»еһӢж–ӯиЁҖиҝҷз§Қж–№ејҸеҸҜд»Ҙе‘ҠиҜүзј–иҜ‘еҷЁпјҢвҖңзӣёдҝЎжҲ‘пјҢжҲ‘зҹҘйҒ“иҮӘе·ұеңЁе№Ід»Җд№ҲвҖқгҖӮ зұ»еһӢж–ӯиЁҖеҘҪжҜ”е…¶е®ғиҜӯиЁҖйҮҢзҡ„зұ»еһӢиҪ¬жҚўпјҢдҪҶжҳҜдёҚиҝӣиЎҢзү№ж®Ҡзҡ„ж•°жҚ®жЈҖжҹҘе’Ңи§Јжһ„гҖӮ е®ғжІЎжңүиҝҗиЎҢж—¶зҡ„еҪұе“ҚпјҢеҸӘжҳҜеңЁзј–иҜ‘йҳ¶ж®өиө·дҪңз”ЁгҖӮ TypeScript дјҡеҒҮи®ҫдҪ пјҢзЁӢеәҸе‘ҳпјҢе·Із»ҸиҝӣиЎҢдәҶеҝ…йЎ»зҡ„жЈҖжҹҘгҖӮ

еңЁtsxиҜӯжі•пјҲreactзҡ„jsxиҜӯжі•зҡ„tsзүҲпјүеҝ…йЎ»йҮҮз”ЁеҗҺйқўдёҖз§Қ

зұ»еһӢж–ӯиЁҖдёҚжҳҜзұ»еһӢиҪ¬жҚўпјҢж–ӯиЁҖжҲҗдёҖдёӘиҒ”еҗҲзұ»еһӢз§ҚдёҚеӯҳеңЁзҡ„зұ»еһӢжҳҜдёҚе…Ғи®ёзҡ„

let num:number|string = "10";
num = 20;
console.log(num.length);//жҠҘй”ҷ
//зұ»еһӢж–ӯиЁҖ  еҸӘиғҪж–ӯиЁҖиҒ”еҗҲзұ»еһӢз§ҚеӯҳеңЁзҡ„зұ»еһӢ
function getAssert(name:string|number){
    //return (<string>name).length;
    return (name as string).length;
    //return (name as boolean).length;//жҠҘй”ҷ
}

зұ»еһӢж–ӯиЁҖжңүдёӨз§ҚеҪўејҸгҖӮ е…¶дёҖжҳҜвҖңе°–жӢ¬еҸ·вҖқиҜӯжі•пјҡ

let someValue: any = 'this is a string'

let strLength: number = (<string>someValue).length

еҸҰдёҖдёӘдёә as иҜӯжі•пјҡ

let someValue: any = 'this is a string'

let strLength: number = (someValue as string).length

дёӨз§ҚеҪўејҸжҳҜзӯүд»·зҡ„гҖӮ иҮідәҺдҪҝз”Ёе“ӘдёӘеӨ§еӨҡж•°жғ…еҶөдёӢжҳҜеҮӯдёӘдәәе–ңеҘҪпјӣз„¶иҖҢпјҢеҪ“дҪ еңЁ TypeScript йҮҢдҪҝз”Ё JSX ж—¶пјҢеҸӘжңү as иҜӯжі•ж–ӯиЁҖжҳҜиў«е…Ғи®ёзҡ„гҖӮ

# зұ»еһӢеҲ«еҗҚ

зұ»еһӢеҲ«еҗҚеҸҜд»Ҙз”ЁжқҘз»ҷдёҖдёӘзұ»еһӢиө·дёҖдёӘж–°еҗҚеӯ—

йҮҮз”Ёе…ій”®еӯ—type

дҫӢеҰӮ:type Name = string | number

дҫӢеӯҗдёӯзҡ„Nameе°ұиЎЁзӨәеҸҜи®ҫзҪ®еӯ—з¬ҰдёІе’Ңж•°еҖјзұ»еһӢ

var str:string|number = "10";
//зұ»еһӢеҲ«еҗҚ
type strType = string|number;
var str:strType ="10";
str = 10;

//еҸҜд»ҘеҜ№дәҺжҺҘеҸЈд№ҹйҮҮз”Ёзұ»еһӢеҲ«еҗҚ
interface muchType1{
    name:string
}

interface muchType2{
    age:number
}

type muchType = muchType1|muchType2;
var obj:muchType = {name:"еј дёү"}
var obj2:muchType = {age:10}
var obj3:muchType = {name:"еј дёү",age:10}

д№ҹеҸҜд»ҘйҮҮз”ЁtypeжқҘзәҰжқҹеҸ–еҖјеҸӘиғҪжҳҜжҹҗдәӣеӯ—з¬ҰдёІдёӯзҡ„дёҖдёӘ

дҫӢеҰӮ:type EventNames = "click"|"scroll"|"mousemove"

//йҷҗеҲ¶еӯ—з¬ҰдёІзҡ„йҖүжӢ©
type sex = "з”·"|"еҘі";
function getSex(s:sex):string{
    return s;
}

getSex("1")//жҠҘй”ҷ
